CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

PROJECT (wxglTerm)

IF(APPLE)
SET(CMAKE_MACOSX_RPATH 1)
SET(CMAKE_PREFIX_PATH /usr/local)
ENDIF()

IF(NOT CMAKE_BUILD_TYPE)
  SET(CMAKE_BUILD_TYPE "DEBUG")
  #SET(CMAKE_BUILD_TYPE "RELEASE")
  #SET(CMAKE_BUILD_TYPE "RELWITHDEBINFO")
  #SET(CMAKE_BUILD_TYPE "MINSIZEREL")
ENDIF()

FIND_PACKAGE(PythonInterp 2.7.13 REQUIRED)
FIND_PACKAGE(PythonLibs 2.7.13 REQUIRED)

FIND_PACKAGE(wxWidgets 3 REQUIRED
  std
  richtext
  )

FIND_PACKAGE(Boost 1.6 REQUIRED)
FIND_PACKAGE(Boost 1.6 REQUIRED python)

SET(Boost_USE_STATIC_LIBS OFF)
SET(Boost_USE_MULTITHREADED ON)
SET(Boost_USE_STATIC_RUNTIME OFF)

INCLUDE_DIRECTORIES("${Boost_INCLUDE_DIRS}" "${PYTHON_INCLUDE_DIRS}")

FILE(GLOB wxglterm_ui_src
    "src/*.h"
    "src/*.cpp"
    )

ADD_DEFINITIONS(-Wall
    )

ADD_LIBRARY(wxglterm_ui SHARED ${wxglterm_ui_src})
SET_TARGET_PROPERTIES(wxglterm_ui PROPERTIES
    PREFIX ""
    )

IF (WIN32)
SET_TARGET_PROPERTIES(wxglterm_ui PROPERTIES
    SUFFIX ".pyd"
    )
ELSE()
SET_TARGET_PROPERTIES(wxglterm_ui PROPERTIES
    SUFFIX ".so"
    )
ENDIF()

TARGET_COMPILE_DEFINITIONS(wxglterm_ui PRIVATE
    ${wxWidgets_DEFINITIONS}
    )
TARGET_INCLUDE_DIRECTORIES(wxglterm_ui PRIVATE
    ${wxWidgets_INCLUDE_DIRS}
    )
TARGET_LINK_LIBRARIES(wxglterm_ui
	${Boost_LIBRARIES}
	${wxWidgets_LIBRARIES}
	${PYTHON_LIBRARIES}
	)
